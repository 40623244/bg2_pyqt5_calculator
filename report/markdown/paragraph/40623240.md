而0按下一次 就只會出現一次

程式碼
 button = self.sender()
        value = int(button.text())
        if self.display.text() == '0' and value == 0.0:
            return
            self.wait =True
        if self.wait :
            self.display.clear()
            self.wait = False
        self.display.setText(self.display.text() + str(value))
